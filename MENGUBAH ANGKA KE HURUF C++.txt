	}
	else if((nilai >60)&&(nilai <80))
	{
		cout << "Nilai Hurufnya Adalah B" << endl;
	}
	else if((nilai <40)&&(nilai <60))
	{
		cout<< "Nilai Hurufnya Adalah C" << endl;
	}
	else if((nilai <20)&&(nilai <40))
	{
		cout <<"Nilai Hurufnya D" << endl;
	}
	else if((nilai <0)&&(nilai <20))

	}
	else

	}

}
